// User Service
// Version 1.0.0

const API_URL = 'https://api.example.com';

function getUser(id) {
    return fetch(`${API_URL}/users/${id}`)
        .then(res => res.json());
}

function createUser(data) {
    return fetch(`${API_URL}/users`, {
        method: 'POST',
        body: JSON.stringify(data),
    }).then(res => res.json());
}

function updateUser(id, data) {
    return fetch(`${API_URL}/users/${id}`, {
        method: 'PUT',
        body: JSON.stringify(data),
    }).then(res => res.json());
}

function deleteUser(id) {
    return fetch(`${API_URL}/users/${id}`, {
        method: 'DELETE',
    });
}

module.exports = {
    getUser,
    createUser,
    updateUser,
    deleteUser,
};
